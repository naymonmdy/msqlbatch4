==>EVENT TYPE (Schedule)
(i) ONE TIME EVENT
(ii) Recurring EVENT

SHOW PROCESSLIST;

SET GLOBAL event_scheduler = ON;
SET GLOBAL event_scheduler = OFF;

-------------------------------------------------------------------------
(i) ONE TIME EVENT

=>METHOD 1
CREATE EVENT eventname
ON Schedule At timestamp
DO 
statements;

=>METHOD 2
CREATE EVENT eventname
ON Schedule At timestamp
ON COMPELSION PRESERVE
DO 
statements;

(ii) Recurring EVENT

=>METHOD 1
CREATE EVENT eventname
ON Schedule EVERY INTERVAL
STARTS timestamp
DO 
statements;

=>METHOD 2
CREATE EVENT eventname
ON Schedule EVERY INTERVAL
STARTS timestamp
ENDS timestamp
DO 
statements;
